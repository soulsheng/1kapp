    public class MyFor{
    　 public static void main(String args[]){
    　　　 outer:
    　　　 for(int i=0;i<2;i++){
    　　　　　 for(int j=0;j<2;j++){
    　　　　　　　 System.out.println("i="+i + " j= "+j);
    　　　　　　　 if(i >0) break outer;

    　 }
